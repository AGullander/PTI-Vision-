  <script>
    const video = document.getElementById('video');
    const scanBtn = document.getElementById('scan-btn');
    const autoBtn = document.getElementById('auto-btn');
    const fileInput = document.getElementById('video-file-input');
    const snapshotImg = document.getElementById('snapshot');
    const bboxLayer = document.getElementById('bbox-layer');
    const errorDiv = document.getElementById('error');
    const containerIdSpan = document.getElementById('container-id');
    const idLockedIndicator = document.getElementById('id-locked-indicator');
    const lastScannedSpan = document.getElementById('last-scanned');
    const statusBadge = document.getElementById('status-badge');
    const detectionsTableBody = document.querySelector('#detections-table tbody');
    const logTableBody = document.querySelector('#log-table tbody');
    const hiddenCanvas = document.getElementById('hidden-canvas');
    const connectionStatus = document.getElementById('connection-status');

    const BACKEND_ANALYZE_URL = 'http://127.0.0.1:8000/api/analyze';
    const BACKEND_TRAIN_URL = 'http://127.0.0.1:8000/api/train_marking';

    let lastLogEntries = [];
    let autoRunning = false;
    let lastFrameWidth = 0;
    let lastFrameHeight = 0;
    let lockedContainerId = null;
    let lastDetections = [];

    function sleep(ms) {
      return new Promise(resolve => setTimeout(resolve, ms));
    }

    // Välj videofil
    fileInput.addEventListener('change', (e) => {
      const file = e.target.files[0];
      if (!file) return;
      const url = URL.createObjectURL(file);
      video.src = url;
      video.play();
      errorDiv.textContent = '';
    });

    async function analyzeFrame() {
      if (!video.src) {
        errorDiv.textContent = 'Ingen videofil vald.';
        return;
      }
      if (video.videoWidth === 0 || video.videoHeight === 0) {
        errorDiv.textContent = 'Videon är inte redo. Spela upp eller pausa och försök igen.';
        return;
      }

      scanBtn.disabled = true;
      errorDiv.textContent = 'Analyserar bild...';

      hiddenCanvas.width = video.videoWidth;
      hiddenCanvas.height = video.videoHeight;
      const ctx = hiddenCanvas.getContext('2d');
      ctx.drawImage(video, 0, 0, hiddenCanvas.width, hiddenCanvas.height);

      lastFrameWidth = hiddenCanvas.width;
      lastFrameHeight = hiddenCanvas.height;

      snapshotImg.src = hiddenCanvas.toDataURL('image/jpeg');

      return new Promise((resolve) => {
        hiddenCanvas.toBlob(async (blob) => {
          try {
            const formData = new FormData();
            formData.append('image', blob, 'frame.jpg');

            const res = await fetch(BACKEND_ANALYZE_URL, {
              method: 'POST',
              body: formData
            });

            if (!res.ok) {
              throw new Error('API-svar: ' + res.status);
            }

            const data = await res.json();
            updateUIWithResult(data);
            errorDiv.textContent = '';
          } catch (err) {
            console.error(err);
            errorDiv.textContent = 'Kunde inte analysera bild: ' + err.message;
          } finally {
            scanBtn.disabled = false;
            resolve();
          }
        }, 'image/jpeg', 0.9);
      });
    }

    function drawBoundingBoxes(detections) {
      bboxLayer.innerHTML = '';

      if (!lastFrameWidth || !lastFrameHeight) return;
      const displayWidth = snapshotImg.clientWidth;
      const displayHeight = snapshotImg.clientHeight;
      if (!displayWidth || !displayHeight) return;

      const scaleX = displayWidth / lastFrameWidth;
      const scaleY = displayHeight / lastFrameHeight;

      detections.forEach(det => {
        if (!det.bbox) return;
        const { x, y, w, h } = det.bbox;

        const box = document.createElement('div');
        box.className = 'bbox';
        box.style.left = (x * scaleX) + 'px';
        box.style.top = (y * scaleY) + 'px';
        box.style.width = (w * scaleX) + 'px';
        box.style.height = (h * scaleY) + 'px';

        bboxLayer.appendChild(box);
      });
    }

    function updateUIWithResult(result) {
      const incomingId = result.container_id || 'UNKNOWN';

      // Lås container-ID: första gången vi får ett riktigt ID, behåll det
      if (incomingId !== 'UNKNOWN') {
        if (!lockedContainerId) {
          lockedContainerId = incomingId;
        } else if (lockedContainerId !== incomingId) {
          // om du senare vill hantera flera containrar kan du ändra logik här
          lockedContainerId = incomingId;
        }
      }

      const idToShow = lockedContainerId || incomingId;
      containerIdSpan.textContent = idToShow;
      idLockedIndicator.textContent = lockedContainerId ? '(låst)' : '';

      const detections = result.detections || [];
      const status = result.status || 'unknown';
      const timestamp = result.timestamp || new Date().toISOString();

      lastDetections = detections;

      lastScannedSpan.textContent = new Date(timestamp).toLocaleString();

      statusBadge.textContent = status === 'ok' ? 'OK' : 'AVVIKELSE';
      statusBadge.className = 'status-badge ' + (status === 'ok' ? 'status-ok' : 'status-alert');

      detectionsTableBody.innerHTML = '';
      detections.forEach((det, idx) => {
        const tr = document.createElement('tr');
        tr.classList.add('clickable');

        const tdLabel = document.createElement('td');
        const tdCat = document.createElement('td');
        const tdSev = document.createElement('td');
        const tdConf = document.createElement('td');

        tdLabel.textContent = det.label || '-';
        tdCat.textContent = det.category || '-';
        tdSev.textContent = det.severity || '-';
        tdConf.textContent = det.confidence != null ? (det.confidence * 100).toFixed(1) + '%' : '-';

        tr.appendChild(tdLabel);
        tr.appendChild(tdCat);
        tr.appendChild(tdSev);
        tr.appendChild(tdConf);

        // Träna-läge: dubbelklick på marking
        tr.addEventListener('dblclick', () => {
          if (det.category === 'marking' && det.code) {
            trainMarking(det, idx);
          }
        });

        detectionsTableBody.appendChild(tr);
      });

      drawBoundingBoxes(detections);

      lastLogEntries.unshift({
        timestamp,
        container_id: idToShow,
        status
      });
      lastLogEntries = lastLogEntries.slice(0, 10);

      logTableBody.innerHTML = '';
      lastLogEntries.forEach(entry => {
        const tr = document.createElement('tr');
        const tdTime = document.createElement('td');
        const tdId = document.createElement('td');
        const tdStatus = document.createElement('td');

        tdTime.textContent = new Date(entry.timestamp).toLocaleTimeString();
        tdId.textContent = entry.container_id;
        tdStatus.textContent = entry.status === 'ok' ? 'OK' : 'AVVIKELSE';

        tr.appendChild(tdTime);
        tr.appendChild(tdId);
        tr.appendChild(tdStatus);
        logTableBody.appendChild(tr);
      });
    }

    async function trainMarking(det, idx) {
      const current = det.label || det.code || '';
      const suggestion = current || (det.code ? `Märkning ${det.code}` : '');
      const title = window.prompt(
        `Ny titel för märkning ${det.code} (t.ex. "Human", "Reefer 45R1")`,
        suggestion
      );
      if (!title) return;

      try {
        const res = await fetch(BACKEND_TRAIN_URL, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ code: det.code, title })
        });
        if (!res.ok) {
          throw new Error('API-svar: ' + res.status);
        }
        const data = await res.json();
        // Uppdatera lokalen label direkt
        lastDetections[idx].label = data.title;
        updateUIWithResult({
          container_id: lockedContainerId || 'UNKNOWN',
          status: statusBadge.textContent === 'OK' ? 'ok' : 'alert',
          detections: lastDetections,
          timestamp: new Date().toISOString(),
        });
      } catch (err) {
        console.error(err);
        window.alert('Kunde inte spara träning: ' + err.message);
      }
    }

    async function autoAnalyze() {
      if (!video.src) {
        errorDiv.textContent = 'Ingen videofil vald.';
        return;
      }

      if (autoRunning) {
        autoRunning = false;
        autoBtn.textContent = 'Auto-analysera hela videon';
        errorDiv.textContent = '';
        return;
      }

      autoRunning = true;
      autoBtn.textContent = 'Stoppa auto-analys';
      errorDiv.textContent = 'Auto-analys pågår...';

      if (video.paused) {
        await video.play();
      }

      while (autoRunning && !video.ended) {
        await analyzeFrame();
        await sleep(1000);

        if (video.ended) break;
        if (video.paused) {
          while (autoRunning && video.paused && !video.ended) {
            await sleep(500);
          }
        }
      }

      autoRunning = false;
      autoBtn.textContent = 'Auto-analysera hela videon';
      errorDiv.textContent = '';
    }

    async function checkBackend() {
      try {
        const res = await fetch(BACKEND_ANALYZE_URL, { method: 'OPTIONS' });
        if (res.ok) {
          connectionStatus.textContent = 'Backend: ansluten';
        } else {
          connectionStatus.textContent = 'Backend: ej kontakt';
        }
      } catch (err) {
        connectionStatus.textContent = 'Backend: ej kontakt';
      }
    }

    scanBtn.addEventListener('click', () => { analyzeFrame(); });
    autoBtn.addEventListener('click', autoAnalyze);

    checkBackend();
  </script>
